# 虚拟DOM

虚拟DOM就是用 JS 对象来描述 DOM 节点，由于操作 DOM 是非常消耗性能的，所以当数据发生变化时，通过对比前后的虚拟DOM节点，通过 DOM-diff 算法计算出需要更新的地方，然后去更新需要更新的视图。

# DOM-diff 算法的过程

# 1.更新的过程

  1. 判断 oldVnode 是不是虚拟节点，如果不是则包装成一个空的虚拟节点
  2. 判断 oldVnode 与 newVnode 是不是同一节点
    1. 是同一节点，则进行更新节点的过程
    2. 不是同一节点，则创建 newVnode 的 DOM 节点，并插入到 oldVnode 的 DOM 节点之前，然后删除 oldVnode 的 DOM 节点
export default function patch (oldVnode, newVnode) {
    // 如果旧节点是DOM节点，则包装成虚拟节点
    if (oldVnode.sel === '' || oldVnode.sel === undefined) {
        oldVnode = vnode(oldVnode.tagName.toLowerCase(), {}, [], '', oldVnode)
    }
    // 判断新旧虚拟节点是不是同一个节点
    if (oldVnode.key === newVnode.key && oldVnode.sel === newVnode.sel) {
        patchVnode(oldVnode, newVnode)
    } else {
        const newVnodeElm = createElement(newVnode)
        const oldVnodeElm = oldVnode.elm
        // 将创建的DOM节点上树，旧的DOM节点的父元素调用insertBefore方法，将新的DOM节点插入到旧的DOM节点之前
        if (oldVnodeElm.parentNode && newVnodeElm) {
            oldVnodeElm.parentNode.insertBefore(newVnodeElm, oldVnodeElm)
        }
        // 将旧的DOM节点删除
        oldVnodeElm.parentNode.removeChild(oldVnodeElm)
    }
}

# 2.更新节点的过程

  1. 如果 newVnode 和 oldVnode 是同一个对象则跳过，不做任何操作

  2. 如果 newVnode 和 oldVnode 均为静态节点则跳过，不做任何操作

    • 静态节点：如果一个结点只包含纯文字，没有任何可变的变量，不管数据怎么变化，只要这个节点第一次渲染了，那么它以后永远不会发生变化，例如：
    <p>我是不会变化的文字</p>
    1

  3. 如果 newVnode 是文本节点

    • 如果 newVnode 是文本节点，说明这个节点只包含纯文本，只需看 oldVnode 是不是文本节
      1. oldVnode 是文本节点，则比较两个文本是否相同，如果不通，则将 oldVnode 的文本改成 newVnode 的文本
      2. oldVnode 不是文本节点，直接调用 setTextNode 方法将它改成文本节点，并且文本内容跟 newVnode 的文本相同

  4. 如果 newVnode 是元素节点

    • newVnode包含子节点
      1. oldVnode 也包含子节点，则进行更新子节点的过程
      2. oldVnode 不包含子节点，如果 oldVnode 是文本节点，则将文本清空，并创建 newVnode DOM 节点，并让 oldVnode 的 DOM 节点的父节点添加创建的 DOM 节点
    • newVnode 不包含子节点
      1. oldVnode 包含子节点，清空 oldVnode 的子节点
      2. oldVnode 是文本节点，将文本清空

export default function patchVnode(oldVnode, newVnode) {
    newVnode.elm = oldVnode.elm
    if (oldVnode === newVnode) return
    // newVnode有text没有children
    if (newVnode.text) {
        // 直接将oldVnode.elm的innerText设置为newVnode的text
        if (oldVnode.text !== newVnode.text) {
            oldVnode.elm.innerText = newVnode.text
        }
    } else { // newVnode有children
        if ((oldVnode.children && oldVnode.children.length > 0) && (newVnode.children && newVnode.children.length > 0)) {
            updateChildren(oldVnode.elm, oldVnode.children, newVnode.children)
        } else if (newVnode.children && newVnode.children.length > 0) {
            if (oldVnode.text) oldVnode.elm.innerText = ''
            const newVnodeElm = createElement(newVnode)
            oldVnode.elm.parentNode.appendChild(newVnodeElm)
        } else if (oldVnode.children && oldVnode.children.length > 0){
            // 清空oldVnode.elm
            oldVnode.elm.innerHTML = ''
        } else if (oldVnode.text) {
            oldVnode.elm.innerText = ''
        }
    }
}

# 2.1.比较节点的过程

  1. 比较旧前节点与新前节点，如果相同，对这两个节点进行更新节点的过程，然后旧前指针和新前指针下移，更新旧前节点和新前节点；如果不同，继续尝试下面三种情况。
  2. 比较旧后节点与新后节点，如果相同，对这两个节点进行更新节点的过程，然后旧后指针和新后指针上移，更新旧后节点和新后节点；如果不同，继续尝试下面两种情况。
  3. 比较旧前节点与新后节点，如果相同，对这两个节点进行更新节点的过程，并且需要移动节点，将旧前节点移动到旧后节点的后面，然后旧前指针下移，新后指针上移，更新旧前节点和新后节点；如果不同，继续尝试下面一种情况。
  4. 比较旧后节点与新前节点，如果相同，对这两个节点进行更新节点的过程，并且需要移动节点，将旧后节点移动到旧前节点的前面，然后旧后指针上移，新前指针下移，更新旧后节点和新前节点。

# 2.2.如果以上四种都没有命中，则要进行循环比较的过程

建立一个 oldChildren 的键为 key 值为下标的 oldKeyToIdx map 对象

  1. 如果 oldKeyToIdx 中包含新前节点的 key，说明不是新节点，则对 oldChildren 中对应的节点和新前节点进行更新节点的操作，然后需要移动节点，将 oldChildren 中对应的节点移动到旧前节点的前面，并将该项设为 undefined，表示已经处理完了。
  2. 如果 oldKeyToIdx 中不包含新前节点的 key，说明是新节点，则创建新节点的 DOM 节点，并插入到旧前节点的前面。
  3. 新前指针下移，更新新前节点，继续循环。

# 2.3.当旧前指针大于旧后指针或者新前指针大于新后指针时，循环结束

  1. 新前指针小于等于新后指针，说明 newChildren 中还有剩余的节点，则以新后指针加1对应节点的 DOM 节点为基准，从新前指针开始到新后指针进行遍历，创建 DOM 节点并插入到基准前面。
  2. 旧前指针小宇等于旧后指针，说明 oldChildren 中还有剩余节点，则从旧前指针开始到旧后指针进行遍历，删除对应节点的 DOM 节点。
export default function updateChildren(parentElm, oldCh, newCh) {
    // 旧前指针
    let oldStartIdx = 0
    // 新前指针
    let newStartIdx = 0
    // 旧后指针
    let oldEndIdx = oldCh.length - 1
    // 新后指针
    let newEndIdx = newCh.length - 1
    // 旧前节点
    let oldStartVnode = oldCh[0]
    // 新前节点
    let newStartVnode = newCh[0]
    // 旧后节点
    let oldEndVnode = oldCh[oldEndIdx]
    // 新后节点
    let newEndVnode = newCh[newEndIdx]
    let oldKeyToIdx = null
    while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
        // 略过undefined或null的项
        if (!oldStartVnode || !oldCh[oldStartIdx]) {
            oldStartVnode = oldCh[++oldStartIdx];
        } else if (!oldEndVnode || !oldCh[oldEndIdx]) {
            oldEndVnode = oldCh[--oldEndIdx];
        } else if (checkSameVnode(oldStartVnode, newStartVnode)) {
            console.log('①旧前与新前命中')
            patchVnode(oldStartVnode, newStartVnode)
            oldStartVnode = oldCh[++oldStartIdx]
            newStartVnode = newCh[++newStartIdx]
        } else if (checkSameVnode(oldEndVnode, newEndVnode)) {
            console.log('②旧后与新后命中')
            patchVnode(oldEndVnode, newEndVnode)
            oldEndVnode = oldCh[--oldEndIdx]
            newEndVnode = newCh[--newEndIdx]
        } else if (checkSameVnode(oldStartVnode, newEndVnode)) {
            console.log('③旧前与新后命中')
            patchVnode(oldStartVnode, newEndVnode)
            // 此时要移动节点，将旧前所指向的节点移动到旧后节点的后面
            parentElm.insertBefore(oldStartVnode.elm, oldEndVnode.elm.nextSibling)
            oldStartVnode = oldCh[++oldStartIdx]
            newEndVnode = newCh[--newEndIdx]
        } else if (checkSameVnode(oldEndVnode, newStartVnode)) {
            console.log('④旧后与新前命中')
            patchVnode(oldEndVnode, newStartVnode)
            // 此时要移动节点，将旧后所指向的节点移动到旧前节点的前面
            parentElm.insertBefore(oldEndVnode.elm, oldStartVnode.elm)
            oldEndVnode = oldCh[--oldEndIdx]
            newStartVnode = newCh[++newStartIdx]
        } else {
            // 创建一个oldKeyToIdx映射，缓存olcVnode的key
            if (!oldKeyToIdx) {
                oldKeyToIdx = {}
                for (let i = oldStartIdx; i <= oldEndIdx; i++) {
                    const key = oldCh[i].key
                    if (key) {
                        oldKeyToIdx[key] = i
                    }
                }
            }
            const idxInOld = oldKeyToIdx[newStartVnode.key]
            if (!idxInOld) {
                // 如果在oldCh里找不到当前循环的newCh里的子节点
                // 新增节点并插入到合适的位置
                parentElm.insertBefore(createElement(newStartVnode), oldStartVnode.elm)
            } else {
                // 如果idxInOld不是undefined，则不是全新的项，需要移动节点
                const vnodeToMove = oldCh[idxInOld]
                patchVnode(vnodeToMove, newStartVnode)
                // 把这项设置为undefined，表示已经处理完了
                oldCh[idxInOld] = undefined
                parentElm.insertBefore(vnodeToMove.elm, oldStartVnode.elm)
            }
            // 指针下移，只移动新前节点
            newStartVnode = newCh[++newStartIdx]
        }
    }

    // 新前指针小于等于新后指针时，说明有需要插入的节点
    if (newStartIdx <= newEndIdx) {
        console.log(newStartIdx)
        console.log(newEndIdx)
        console.log('new还有剩余的节点没有处理')
        for (let i = newStartIdx; i <= newEndIdx; i++) {
            const before = newCh[newEndIdx + 1] == null ? null : newCh[newEndIdx + 1].elm
            console.log(before)
            // insertBefore方法可以识别null，如果是null就会自动排到队尾去
            // 此时的newCh[i]还不是真正的DOM，要调用createElement函数转为真实DOM
            parentElm.insertBefore(createElement(newCh[i]), before)
        }
    } else if (oldStartIdx <= oldEndIdx) {
        console.log('old还有剩余的节点没有处理')
        // 批量删除oldStartIdx和oldEndIdx之间的项
        for (let i = oldStartIdx; i <= oldEndIdx; i++) {
            if (oldCh[i]) {
                parentElm.removeChild(oldCh[i].elm)
            }
        }
    }
}

function checkSameVnode(a, b) {
    return a.sel === b.sel && a.key === b.key
}

# DOM-diff 常见问题

# 1.DOM-diff 算法时间复杂度

一般的 DOM-diff 时间复杂度是 O(n^3)，但是 Vue 只进行同层级比较，不进行跨层级比较，所以可以优化到 O(n)

# 2.为什么有了数据劫持还需要 DOM-diff？

Vue 实现双向绑定的时候，会给指令对应的节点绑定一个 watcher，如果绑定的细粒度过高，就会产生大量的 watcher，这会带来内存以及依赖追踪的开销，细粒度过低则无法精准的侦测变化。所以 Vue 采用中等细粒度的方案，在组件级别采用 watcher，在组件内部采用 DOM-diff 算法进行更新。

# 3.Vue 中的 key 有什么用？

key 是 Vue 中 vnode 的唯一标识，通过这个 key，diff 操作可以更准确、迅速。

如果不使用 key 或者使用 index 作为 key，那么 Vue 在更新的时候会采用就地复用的策略，对新旧节点进行额外的 patch 操作，如果使用了唯一的 key，则只会进行移动节点的操作。
